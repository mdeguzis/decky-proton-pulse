# i18n System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a lightweight internationalization system supporting 10 languages with auto-detection, typed nested keys, template functions for pluralization, and English fallbacks.

**Architecture:** A single `src/lib/i18n.ts` module contains all types, translations, detection logic, and a reactive `t()` accessor. Language preference is stored in localStorage via the existing `settings.ts` helpers. A `useLanguage()` React hook triggers re-renders on language change.

**Tech Stack:** TypeScript, React hooks, Vitest, existing `settings.ts` (localStorage)

const STEAM_LANG_MAP: Record<string, Language> = {
  english: 'en',
  schinese: 'zh-CN',
  russian: 'ru',
  brazilian: 'pt-BR',
  german: 'de',
  spanish: 'es',
  french: 'fr',
  japanese: 'ja',
  koreana: 'ko',
  turkish: 'tr',

// ── Detection ─────────────────────────────────────────────────────────────────

export function detectLanguage(): Language {
  // 1. Steam UI language
  try {
    const steamLang: string | undefined =
      (globalThis as any).SteamClient?.Settings?.GetCurrentLanguage?.();
    if (steamLang && STEAM_LANG_MAP[steamLang]) return STEAM_LANG_MAP[steamLang];
  } catch { /* ignore */ }

  // 2. navigator.language
  try {
    const navLang = globalThis.navigator?.language;
    if (navLang) {
      const exact = LANGUAGES.find((l) => l.toLowerCase() === navLang.toLowerCase());
      if (exact) return exact;
      const prefix = navLang.split('-')[0].toLowerCase();
      const prefixMatch = LANGUAGES.find((l) => l.split('-')[0].toLowerCase() === prefix);
      if (prefixMatch) return prefixMatch;
    }
  } catch { /* ignore */ }

  // 3. Fallback
  return 'en';
}

// ── Translations registry (start with English only, others added in Task 2) ──

const translations: Record<Language, TranslationTree | null> = {
  en,
  'zh-CN': null,
  ru: null,
  'pt-BR': null,
  de: null,
  es: null,
  fr: null,
  ja: null,
  ko: null,
  tr: null,
};

export function registerTranslation(lang: Language, tree: TranslationTree): void {
  translations[lang] = tree;
}

// ── Active language state ─────────────────────────────────────────────────────

let activeLang: Language | 'auto' = getSetting<Language | 'auto'>('language', 'auto');
let resolvedLang: Language = activeLang === 'auto' ? detectLanguage() : activeLang;
let version = 0; // bumped on every setLanguage() call

export function setLanguage(lang: Language | 'auto'): void {
  activeLang = lang;
  setSetting('language', lang);
  resolvedLang = lang === 'auto' ? detectLanguage() : lang;
  version++;
  // Notify React subscribers
  for (const cb of listeners) cb();
}

export function getActiveLanguage(): Language {
  return resolvedLang;
}

export function getLanguageVersion(): number {
  return version;
}

// ── React hook ────────────────────────────────────────────────────────────────

type Listener = () => void;
const listeners = new Set<Listener>();

function subscribe(cb: Listener): () => void {
  listeners.add(cb);
  return () => { listeners.delete(cb); };
}

// import { useSyncExternalStore } from 'react';
// We can't import React at module level in a lib file without bundler issues,
// so useLanguage is exported as a function that the consuming component calls.
// Components must import React themselves.

export function useLanguage(): Language {
  // Using React's useSyncExternalStore for tear-free reads
  const { useSyncExternalStore } = require('react');
  return useSyncExternalStore(subscribe, getActiveLanguage, getActiveLanguage);
}

// ── t() accessor with fallback ────────────────────────────────────────────────

const IS_DEV = typeof process !== 'undefined' && process.env?.NODE_ENV !== 'production';

function withFallback(lang: Language): TranslationTree {
  const tree = translations[lang];
  if (!tree || lang === 'en') return en;
  // Build a proxy that falls back to English for missing keys
  return buildFallbackProxy(tree, en, lang, '') as TranslationTree;
}

function buildFallbackProxy(target: any, fallback: any, lang: Language, path: string): any {
  return new Proxy(target, {
    get(obj, prop: string) {
      const val = obj[prop];
      const fbVal = fallback[prop];
      if (val === undefined || val === null) {
        if (IS_DEV && typeof fbVal === 'string') {
          // Dynamically import would create a dep cycle; inline the warning
          console.warn(`[i18n] Missing translation: ${lang}.${path}${String(prop)}`);
          return `[!] ${fbVal}`;
        }
        return fbVal;
      }
      if (typeof val === 'object' && val !== null && typeof fbVal === 'object' && fbVal !== null) {
        return buildFallbackProxy(val, fbVal, lang, `${path}${String(prop)}.`);
      }
      return val;
    },
  });
}

export function t(): TranslationTree {
  return withFallback(resolvedLang);
}

describe('translation completeness', () => {
  it('all registered languages have the same top-level keys as English', () => {
    const enTree = t();
    const enKeys = Object.keys(enTree).sort();

    for (const lang of LANGUAGES) {
      if (lang === 'en') continue;
      setLanguage(lang);
      const tree = t();
      const keys = Object.keys(tree).sort();
      expect(keys).toEqual(enKeys);
    }

    setLanguage('auto');
  });
});

describe('fallback proxy', () => {
  it('returns English string for missing keys in non-English language', () => {
    // Even if a language has gaps, t() should never return undefined
    for (const lang of LANGUAGES) {
      setLanguage(lang);
      const tree = t();
      expect(typeof tree.common.save).toBe('string');
      expect(tree.common.save.length).toBeGreaterThan(0);
      expect(typeof tree.reports.found(3)).toBe('string');
    }
    setLanguage('auto');
  });
});

describe('pluralization', () => {
  it('Russian reports.found handles plural forms correctly', () => {
    setLanguage('ru');
    const tree = t();
    // 1 → singular
    expect(tree.reports.found(1)).toContain('1');
    // 5 → genitive plural
    expect(tree.reports.found(5)).toContain('5');
    // 21 → singular again
    expect(tree.reports.found(21)).toContain('21');
    setLanguage('auto');
  });

  it('Japanese reports.found works without plural forms', () => {
    setLanguage('ja');
    const tree = t();
    expect(tree.reports.found(1)).toContain('1');
    expect(tree.reports.found(5)).toContain('5');
    setLanguage('auto');
  });
});
